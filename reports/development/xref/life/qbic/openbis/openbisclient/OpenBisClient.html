111    * Function map an integer value to a char
112    *
113    * @param i the integer value which should be mapped
114    * @return the resulting char value
115    */
116   public static char mapToChar(int i) {
117     i += 48;
118     if (i > 57) {
119       i += 7;
120     }
121     return (char) i;
122   }

124   /**
125    * Function to generate the checksum for the given barcode string
126    *
127    * @param s the barcode string
128    * @return the checksum for the given barcode
129    */
130   public static char checksum(String s) {
131     int i = 1;
132     int sum = 0;
133     for (int idx = 0; idx <= s.length() - 1; idx++) {
134       sum += (((int) s.charAt(idx))) * i;
135       i += 1;
136     }
137     return mapToChar(sum % 34);
138   }

559   /**
560    * Returns Space names a given user should be able to see
561    *
562    * @param userID Username found in openBIS
563    * @return List of space names with projects this user has access to
564    */
565   @Override
566   public List<String> getUserSpaces(String userID) {
567     // this sets the user sessionToken
568     loginAsUser(userID);
569     List<String> spaceIdentifiers = new ArrayList<>();
570     // we are not using external functions to make sure this user is actually used
571     try {
572       SearchResult<Space> spaces =
573           v3.searchSpaces(sessionToken, new SpaceSearchCriteria(), new SpaceFetchOptions());
574       if (spaces != null) {
575         for (Space space : spaces.getObjects()) {
576           spaceIdentifiers.add(space.getCode());
577         }
578       }
579     } catch (UserFailureException u) {
580       logger.error("Could not fetch spaces for user " + userID
581           + ", because they could not be logged in. Is user " + this.userId + " an admin user?");
582       logger.warn("No spaces were returned.");
583     }
584     logout();
585     login();
586 
587     return spaceIdentifiers;
588   }

590   /**
591    * Returns wether a user is instance admin in openBIS
592    *
593    * @return true, if user is instance admin, false otherwise
594    */
595   @Override
596   public boolean isUserAdmin(String userID) {
597     ensureLoggedIn();
598     PersonSearchCriteria criteria = new PersonSearchCriteria();
599     criteria.withUserId().thatEquals(userID);
600     PersonFetchOptions options = new PersonFetchOptions();
601     options.withRoleAssignments();
602     SearchResult<Person> res = v3.searchPersons(sessionToken, criteria, options);
603     for (Person p : res.getObjects()) {
604       for (RoleAssignment r : p.getRoleAssignments()) {
605         if (r.getRole().equals(Role.ADMIN)) {
606           return true;
607         }
608       }
609     }
610     return false;
611   }

1026   /**
1027    * Returns lines of a spreadsheet of humanly readable information of the samples in a project.
1028    * Only one requested layer of the data model is returned. Experimental factors are returned in
1029    * the properties xml format and should be parsed before the spreadsheet is presented to the user.
1030    *
1031    * @param projectCode The 5 letter QBiC code of the project
1032    * @param sampleType The openBIS sampleType that should be included in the result
1033    * @return List containing each line of the resulting TSV fil
1034    */
1035   public List<String> getProjectTSV(String projectCode, String sampleType) {
1036     List<String> res = new ArrayList<String>();
1037     // search all samples of project
1038     List<Sample> allSamples = getSamplesOfProject(projectCode);
1039     // filter all samples by types
1040     List<Sample> samples = new ArrayList<Sample>();
1041     for (Sample s : allSamples) {
1042       if (sampleType.equals(s.getType().getCode()))
1043         samples.add(s);
1044     }
1045     Map<String, String> ncbi = new HashMap<>();
1046     Map<String, String> taxonomyMap = getVocabCodesAndLabelsForVocab("Q_NCBI_TAXONOMY");
1047     for (String key : taxonomyMap.keySet()) {
1048       ncbi.put(taxonomyMap.get(key), key);
1049     }
1050     String header = "QBiC Code\tSecondary Name\tLab ID\tSample Type\tAttributes\tSource";
1051     if (!sampleType.equals("Q_BIOLOGICAL_ENTITY"))
1052       header += "\tSource Name(s)\tSource Lab ID(s)";
1053     if (sampleType.equals("Q_TEST_SAMPLE"))
1054       header += "\tExtract Code(s)\tExtract Name(s)\tExtract Lab ID(s)";
1055     res.add(header);
1056     for (Sample sample : samples) {
1057       String code = sample.getCode();
1058       List<String> row = new ArrayList<String>();
1059       row.add(code);
1060       String secName = getSamplePropertyOrEmptyString(sample, "Q_SECONDARY_NAME");
1061       row.add(secName);
1062       String extID = getSamplePropertyOrEmptyString(sample, "Q_EXTERNALDB_ID");
1063       row.add(extID);
1064       String extrType = getSamplePropertyOrEmptyString(sample, "Q_PRIMARY_TISSUE");
1065       if (extrType.isEmpty()) {
1066         extrType = getSamplePropertyOrEmptyString(sample, "Q_SAMPLE_TYPE");
1067       }
1068       if (extrType.equals("CELL_LINE")) {
1069         extrType = sample.getProperties().get("Q_TISSUE_DETAILED");
1070       }
1071       row.add(extrType);
1072       String props = sample.getProperties().get("Q_PROPERTIES");
1073       if (props == null)
1074         props = "<?xml";
1075       row.add(props);
1076       row.add(fetchSource(new ArrayList<Sample>(Arrays.asList(sample)), ncbi));
1077       if (!sampleType.equals("Q_BIOLOGICAL_ENTITY")) {
1078         Set<Sample> sources = fetchAncestorsOfType(new ArrayList<Sample>(Arrays.asList(sample)),
1079             "Q_BIOLOGICAL_ENTITY");
1080         row.add(getPropertyOfSamples(sources, "Q_SECONDARY_NAME"));
1081         row.add(getPropertyOfSamples(sources, "Q_EXTERNALDB_ID"));
1082       }
1083       if (sampleType.equals("Q_TEST_SAMPLE")) {
1084         Set<Sample> extracts = fetchAncestorsOfType(new ArrayList<Sample>(Arrays.asList(sample)),
1085             "Q_BIOLOGICAL_SAMPLE");
1086         List<String> codeL = new ArrayList<String>();
1087         for (Sample s : extracts) {
1088           codeL.add(s.getCode());
1089         }
1090         row.add(org.apache.commons.lang3.StringUtils.join(codeL, ", "));
1091         row.add(getPropertyOfSamples(extracts, "Q_SECONDARY_NAME"));
1092         row.add(getPropertyOfSamples(extracts, "Q_EXTERNALDB_ID"));
1093       }
1094       res.add(org.apache.commons.lang3.StringUtils.join(row, "\t"));
1095     }
1096     return res;
1097   }

1137   /**
1138    * used by getProjectTSV to get the Patient(s)/Source(s) of a list of samples
1139    * 
1140    * @param samples List of openbis samples
1141    * @param ncbi List of openbis vocabulary terms of the NCBI taxonomy vocabulary
1142    * @return String representation of the root source(s) of thse samples
1143    */
1144   protected static String fetchSource(List<Sample> samples, Map<String, String> ncbi) {
1145     List<String> res = new ArrayList<String>();
1146     boolean isCellLine = false;
1147     Set<Sample> roots = new HashSet<Sample>();
1148     while (!samples.isEmpty()) {
1149       Set<Sample> store = new HashSet<Sample>();
1150       for (Sample sample : samples) {
1151         if (!sample.getType().getCode().equals("Q_BIOLOGICAL_ENTITY")) {
1152           store.addAll(sample.getParents());
1153           if (sample.getType().getCode().equals("Q_BIOLOGICAL_SAMPLE"))
1154             isCellLine = sample.getProperties().get("Q_PRIMARY_TISSUE").equals("CELL_LINE");
1155         } else
1156           roots.add(sample);
1157       }
1158       samples = new ArrayList<Sample>(store);
1159     }
1160     for (Sample sample : roots) {
1161       String id = sample.getCode();
1162       try {
1163         id = id.split("-")[1];
1164       } catch (ArrayIndexOutOfBoundsException e) {
1165       }
1166       String organism = sample.getProperties().get("Q_NCBI_ORGANISM");
1167       if (organism != null) {
1168         String desc = "";
1169         for (String code : ncbi.keySet()) {
1170           if (organism.equals(code))
1171             desc = ncbi.get(code);
1172         }
1173         if (isCellLine)
1174           desc += " Cell Line";
1175         else if (desc.toLowerCase().equals("homo sapiens"))
1176           desc = "Patient";
1177         res.add(desc + ' ' + id);
1178       } else
1179         res.add("unknown source");
1180     }
1181     String[] resArr = new String[res.size()];
1182     resArr = res.toArray(resArr);
1183     return StringUtils.join(resArr, "+");
1184   }
