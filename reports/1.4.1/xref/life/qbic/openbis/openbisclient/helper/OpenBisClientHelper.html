107   public static List<PropertyType> getPropertiesOfEntityType(IEntityType type) {
108   List<PropertyType> res = new ArrayList<>();
109   List<PropertyAssignment> assignments = type.getPropertyAssignments();
110   for (PropertyAssignment propertyAssignment : assignments) {
111     res.add(propertyAssignment.getPropertyType());
112   }
113   return res;
